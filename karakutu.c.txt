void loop()
{
	    
	    while(serialDataAvail(fd)) // uart datasý geldiðinde iþletilir, uart kesmesi
	    {
			//printf("serialdataavail fd:%d\n",fd);
			sleep(1);
			txt[0]=serialGetchar(fd); // ilk karekter alýnýyor. $ ise tüm karekterleri \n ifadesidesine kadar alýyoruz. sonraki karakter döndürülür
		
		
			if (txt[0]=='$')  
			{
				for (i=1;i<100;i++)
				{
					txt[i] = serialGetchar(fd);
					if (txt[i]=='\n') break;						
				}					
			}
	   
			if ( txt[3]=='G' && txt[4]=='G' && txt[5]=='A')
			{	//3.4.5. karaterle GGA ise gga deðiþkenine kopyalanýyor.																						
																								
				memcpy(gga, txt, strlen(txt)+1);
				parcala_GGA();																															
																																																																								
			}
			
			if ( txt[3]=='R' && txt[4]=='M' && txt[5]=='C')	
			{   //3.4.5. karaterle RMC ise rmc deðiþkenine kopyalanýyor.
															
				memcpy(rmc, txt, strlen(txt)+1);
				parcala_RMC();														
			}											  
		}
}

	
int  main() 
{	
	wiringPiSetup ();	// Gpio kütüphanesi kuruluyor
	pinMode(1, OUTPUT); // dosya kaydý ledi çýkýþ olarak ayarlanýyor
	
	fd = serialOpen ("/dev/ttyS0", 9600); // 9600 bund hýzýnda seri iletiþim ayarlanýyor.

	signal(SIGINT, exit_event); 					// ctrl+c çýkýþý için olay döngüsü ctrl+c basýldýgýnda exit_event iþletilecek program çýkýþý bu þekilde olmalýdýr.
	pthread_t pth;									// thread fonksitonlarý oluþturuluyor.	
	pthread_create(&pth,NULL,th,"processing..."); 	// GGA.fix deðiþkeni 1 veya 2 ise dosya yazmak için sürekli çalýþýr.
	pthread_create(&pth,NULL,led,"processing..."); 	// dosya yazýlýyorsa led flash
 
	printf (" -------Uygulama baþlatýlýyor----\n") ;
   
	while(1) loop();

	pthread_join(pth, NULL );       
return 0;      	      
}
